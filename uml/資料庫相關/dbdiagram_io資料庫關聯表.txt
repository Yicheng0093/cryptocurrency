Table auth_user {
  id int [pk]
  password varchar(128)
  last_login datetime
  is_superuser bool
  username varchar(150)
  first_name varchar(150)
  last_name varchar(150)
  email varchar(254)
  is_staff bool
  is_active bool
  date_joined datetime
}

  has_seen_know_modal bool
}
